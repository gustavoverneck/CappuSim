üåü Custom Initial Conditions

You can define custom initial conditions by modifying the lbm.set_condition() function in main.rs. This function allows you to set the initial velocity, density, and flags for your simulation. Tailor these values to match your specific use case or experiment. Example (2D Von-K√°rm√°n Vortex): 

      fn main() {
        let nx = 256;
        let ny = 128;

        let viscosity = 0.01;
        let omega = 1.0 / (3.0 * viscosity + 0.5);
        let u0 = 0.1;

        // Initialize LBM simulation
        let mut lbm = LBM::new(nx, ny, 1, "D2Q9".to_string(), viscosity);

        // Cylinder parameters
        let radius = nx as f32 * 0.08;
        let cx = nx as i32 / 4; // 25% from left
        let cy = ny as i32 / 2;

        lbm.set_conditions(|lbm, x, y, _z, n| {
          let dx = x as i32 - cx;
          let dy = y as i32 - cy;
          let dist = ((dx * dx + dy * dy) as f32).sqrt();

          if dist <= radius {
            lbm.flags[n] = FLAG_SOLID; // Cylinder obstacle
          } else if x == 0 {
            // Inlet with prescribed velocity
            lbm.flags[n] = FLAG_EQ;
            lbm.velocity[n].x = u0;
            lbm.velocity[n].y = 0.0;
            lbm.density[n] = 1.0;
          } else if x == nx - 1 {
            // Outflow
            lbm.flags[n] = FLAG_EQ;
            lbm.velocity[n].x = u0;
            lbm.velocity[n].y = 0.0;
            lbm.density[n] = 1.0;
          } else if y == 0 || y == ny - 1 {
            // Top and bottom walls
            lbm.flags[n] = FLAG_SOLID;
          } else {
            // Normal fluid region
            lbm.flags[n] = FLAG_FLUID;
            lbm.velocity[n].x = u0;
            lbm.velocity[n].y = 0.0;
            lbm.density[n] = 1.0;
          }
        });

        lbm.set_output_vtk(true);
        lbm.set_output_interval(50);
        lbm.run(10000);
      }

üßæ Output Format

Results are saved in output/data_XXXX.csv or output/data_XXXX.vtk. You can activate output by typing: 

      // to enable csv output
        lbm.set_output_csv(true);

      // to enable vtk output
      lbm.set_output_vtk(true);

      // to set output interval
      lbm.set_output_interval(integer_interval);
Each file contains: coordinates, density, velocity, viscosity , q-criterion
